import collections

def level_order(root):
    if not root:
        return []
    res, deque = [], collections.deque([root])
    while deque:
        tmp = []
        for _ in range(len(deque)):
            node = deque.popleft()
            tmp.append(node.val)
            if node.left:
                deque.append(node.left)
            if node.right:
                deque.append(node.right)
        res.append(tmp)
    return res
