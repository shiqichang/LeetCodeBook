```python
class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        hash_map = dict()
        nums_cp = copy.deepcopy(nums)
        for index, elem in enumerate(nums_cp):
            if elem in hash_map:
                nums.remove(elem)
            else:
                hash_map[elem] = index
        return len(nums)
```

```python
class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        if not nums:
            return 0

        i = 0
        for j in range(1, len(nums)):
            if nums[j] != nums[i]:
                i += 1
                nums[i] = nums[j]
        return i + 1
```
